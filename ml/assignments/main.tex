\section*{2) Multiple-Choice Questions (MCQ)}
\begin{enumerate}[label=\textbf{MCQ \arabic*.}]
  \item In k-NN, the main "training" phase is:
  \begin{enumerate}[label=(\Alph*)]
    \item Gradient descent optimization
    \item Tree pruning
    \item Storing indexed labeled samples
    \item Backpropagation
  \end{enumerate}

  \item If feature scales are very different, the most likely issue is:
  \begin{enumerate}[label=(\Alph*)]
    \item Data leakage disappears
    \item Distance becomes dominated by large-scale features
    \item Overfitting is impossible
    \item $k$ has no effect
  \end{enumerate}

  \item Increasing $k$ in k-NN generally tends to:
  \begin{enumerate}[label=(\Alph*)]
    \item Increase variance and reduce bias
    \item Reduce variance and increase bias
    \item Reduce both bias and variance always
    \item Have no effect
  \end{enumerate}

  \item Which metric compares vector direction rather than magnitude?
  \begin{enumerate}[label=(\Alph*)]
    \item Euclidean distance
    \item Manhattan distance
    \item Cosine similarity
    \item Hamming distance
  \end{enumerate}

  \item Which statement is true for brute-force k-NN query cost?
  \begin{enumerate}[label=(\Alph*)]
    \item Typically scales with $O(nd)$ per query
    \item Always constant time
    \item Independent of feature dimension
    \item Always faster than ANN
  \end{enumerate}

  \item The main purpose of a validation set is:
  \begin{enumerate}[label=(\Alph*)]
    \item To replace the training set entirely
    \item To tune hyperparameters such as $k$ without touching the final test set
    \item To store unlabeled data only
    \item To guarantee zero generalization gap
  \end{enumerate}

  \item In a retrieval setting, a common metric is:
  \begin{enumerate}[label=(\Alph*)]
    \item BLEU only
    \item ROC-AUC only
    \item Recall@k
    \item MSE only
  \end{enumerate}

  \item If training error is near zero but validation error is much larger for $k=1$, the best interpretation is:
  \begin{enumerate}[label=(\Alph*)]
    \item The model is underfitting
    \item The distance metric is irrelevant
    \item The model is likely overfitting
    \item The dataset no longer needs a test split
  \end{enumerate}

  \item A small generalization gap means:
  \begin{enumerate}[label=(\Alph*)]
    \item Training and unseen-data performance are relatively close
    \item The model must be linear
    \item The hypothesis class has size one
    \item There is no noise in the data
  \end{enumerate}
\end{enumerate}

\section*{4) Implementation Tasks (NumPy from Scratch)}

\medskip
\textbf{Core equations to use (reference):}
\begin{itemize}
  \item For two vectors $x, z \in \mathbb{R}^d$, Euclidean distance is
  \[
    d_{L_2}(x,z)=\sqrt{\sum_{j=1}^{d}(x_j-z_j)^2}.
  \]
  \item Manhattan distance is
  \[
    d_{L_1}(x,z)=\sum_{j=1}^{d}|x_j-z_j|.
  \]
  \item Cosine similarity is
  \[
    \cos(x,z)=\frac{x^\top z}{\|x\|_2\|z\|_2},
  \]
  and in this assignment cosine distance is
  \[
    d_{\cos}(x,z)=1-\cos(x,z).
  \]
  \item For classification, uniform-vote prediction is the majority label among the $k$ nearest neighbors:
  \[
    \hat y = \arg\max_{c} \sum_{i \in \mathcal{N}_k(x)} \mathbf{1}[y_i=c].
  \]
  \item For distance-weighted classification, use weights
  \[
    w_i=\frac{1}{d(x,x_i)+10^{-12}}.
  \]
  \item For regression, uniform prediction is the mean of neighbor targets:
  \[
    \hat y = \frac{1}{k}\sum_{i \in \mathcal{N}_k(x)} y_i.
  \]
  \item For distance-weighted regression, use
  \[
    \hat y = \frac{\sum_{i \in \mathcal{N}_k(x)} w_i y_i}{\sum_{i \in \mathcal{N}_k(x)} w_i}.
  \]
  \item Standardize features using training-set statistics only:
  \[
    x'_j = \frac{x_j-\mu_j}{\sigma_j}.
  \]
\end{itemize}

\end{itemize}

\medskip
\textbf{Core PCA equations to use and reference:}
\begin{itemize}
  \item Mean and centered data:
  \[
    \mu = \frac{1}{n}\sum_{i=1}^{n}x_i,\qquad X_c = X-\mathbf{1}\mu^\top.
  \]
  \item Covariance:
  \[
    \Sigma=\frac{1}{n-1}X_c^\top X_c.
  \]
  \item First component objective:
  \[
    w_1=\arg\max_{\|w\|_2=1} w^\top \Sigma w.
  \]
  \item Projection and reconstruction:
  \[
    z = U_k^\top(x-\mu),\qquad \hat{x}=\mu+U_k z.
  \]
  \item Reconstruction error anomaly score:
  \[
    a(x)=\|x-\hat{x}\|_2^2.
  \]
\end{itemize}
