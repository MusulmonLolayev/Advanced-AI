\section{Principal Component Analysis (PCA)}

\begin{frame}{From Supervised to Unsupervised Learning}
\begin{itemize}
  \pause
  \item In Lecture 1, the focus was supervised learning, especially classification with labeled data.
  \pause
  \item In this lecture, the focus shifts to unsupervised learning, where the goal is to discover structure without target labels.
  \pause
  \item Principal Component Analysis (PCA) is a central unsupervised method for finding dominant directions of variation in data.
  \pause
  \item PCA provides a bridge from raw high-dimensional features to lower-dimensional representations used for visualization, compression, and preprocessing.
\end{itemize}
\end{frame}

\begin{frame}{Lecture Outline}
\begin{itemize}
  \pause
  \item Motivate PCA as a response to redundancy, correlation, and high-dimensional representations.
  \pause
  \item Start from projection and derive variance maximization.
  \pause
  \item Connect the optimization problem to eigenvectors, eigenvalues, and orthogonality.
  \pause
  \item Extend the discussion to multiple components, reconstruction, and explained variance.
  \pause
  \item Conclude with practical computation, applications, and how PCA connects to the lab.
\end{itemize}
\end{frame}

\begin{frame}{Why PCA in Unsupervised Learning?}
\begin{itemize}
  \item In supervised learning, labels indicate what should be predicted.
  \pause
  \item In unsupervised learning, labels are absent, so the task is to analyze the structure of the input data itself.
  \pause
  \item PCA addresses this by identifying directions that capture the largest variation in the data.
\end{itemize}

\pause
\begin{rlintuition}
PCA identifies the directions in the data that carry the greatest variation.
\end{rlintuition}
\end{frame}

\begin{frame}{PCA Versus Raw Features}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{With raw features}
\begin{itemize}
  \item Distances may be distorted by noise or redundancy.
  \pause
  \item Correlated features may repeat the same information.
  \pause
  \item High-dimensional data are hard to visualize and expensive for k-NN.
\end{itemize}

\pause
\column{0.5\textwidth}
\textbf{With PCA features}
\begin{itemize}
  \item Retain a smaller set of directions ranked by variance.
  \pause
  \item Often get faster distance computation and cleaner geometry.
  \pause
  \item Enable 2D visualization, compression, denoising, and reconstruction-based anomaly detection.
\end{itemize}
\end{columns}

\pause
\vspace{0.35em}
\footnotesize PCA can be useful, but it does not guarantee improved prediction: low-variance directions may still matter for the target.
\end{frame}

\begin{frame}{Centering the Data Before PCA}
\begin{itemize}
  \item PCA studies variation around the mean, so centering is essential.
  \pause
  \item For training examples $x_1,\dots,x_n \in \mathbb{R}^d$, compute the sample mean
  \[
    \bar{x}=\frac{1}{n}\sum_{i=1}^n x_i.
  \]
  \pause
  \item Then define centered examples
  \[
    x_i^c = x_i-\bar{x},
    \qquad
    X_c =
    \begin{bmatrix}
      (x_1^c)^\top\\
      \vdots\\
      (x_n^c)^\top
    \end{bmatrix}.
  \]
  \pause
  \item From this point onward, $x_i$ denotes a centered example unless stated otherwise.
  \pause
  \item Without centering, the first principal direction can be influenced by the mean level rather than by true variation.
  \pause
  \item \footnotesize Example: if $x_1=(4,2)$ and $x_2=(2,0)$, then $\bar{x}=(3,1)$ and the centered examples are $(1,1)$ and $(-1,-1)$.
\end{itemize}
\end{frame}

\begin{frame}{Data Matrix and Projection}
\begin{columns}[T]
\column{0.55\textwidth}
\begin{itemize}
  \item Stack centered training examples in $X_c \in \mathbb{R}^{n\times d}$.
  \pause
  \item Choose a direction $w \in \mathbb{R}^d$.
  \pause
  \item Each example projects to one scalar $z_i = w^\top x_i$.
  \pause
  \item For all training examples, $z = X_c w \in \mathbb{R}^n$ collects the projected coordinates.
  \pause
  \item PCA asks which direction keeps the largest spread after this projection.
  \pause
  \item \footnotesize Example: if $x=(2,1)$ and $w=(1,0)$, then $z=w^\top x=2$.
\end{itemize}

\pause

\begin{frame}{Projected Variance: Scalar Form}
\begin{itemize}
  \item The projected coordinates are $z_i = w^\top x_i$.
  \pause
  \item Because the centered examples have mean zero, the projected mean is also zero.
  \pause
  \item Start from the sample variance of the projected coordinates:
  \begin{equation}
    \mathrm{Var}(w^\top x)
    = \frac{1}{n-1}\sum_{i=1}^{n}(w^\top x_i)^2.
    \label{eq:proj-var-pca}
  \end{equation}
  \pause
  \item Use the identity
  \[
    (w^\top x_i)^2 = w^\top x_i x_i^\top w,
  \]
  which converts each squared scalar projection into a quadratic form.
\end{itemize}
\end{frame}

\begin{frame}{Projected Variance: Matrix Form}
\begin{itemize}
  \item Substitute the identity into the variance expression:
  \[
    \mathrm{Var}(w^\top x)
    = \frac{1}{n-1}\sum_{i=1}^{n} w^\top x_i x_i^\top w
    = w^\top \left(\frac{1}{n-1}\sum_{i=1}^{n} x_i x_i^\top\right) w.
  \]
  \pause
  \item Define the covariance matrix by
  \[
    \Sigma = \frac{1}{n-1}X_c^\top X_c
           = \frac{1}{n-1}\sum_{i=1}^{n} x_i x_i^\top,
  \]
  so the projected variance becomes
  \[
    \mathrm{Var}(w^\top x)=w^\top \Sigma w.
  \]
  \pause
  \item PCA therefore searches for the direction that maximizes the quadratic form $w^\top \Sigma w$.
\end{itemize}
\end{frame}

\begin{frame}{PCA as a Constrained Problem}
\begin{itemize}
  \item The first principal direction solves
  \begin{equation}
    \max_{w} \; w^\top \Sigma w
    \quad \text{subject to} \quad
    w^\top w = 1.
    \label{eq:pca-constrained}
  \end{equation}
  \pause
  \item The unit-norm constraint fixes scale.
  \pause
  \item Without it, multiplying $w$ by a constant would change the objective artificially.
\end{itemize}
\end{frame}

\begin{frame}{From Optimization to Eigenvectors}
\begin{itemize}
  \item Introduce a Lagrange multiplier $\lambda$:
  \begin{equation}
    \mathcal{L}(w,\lambda)=w^\top \Sigma w-\lambda (w^\top w-1).
    \label{eq:lagrangian}
  \end{equation}
  \pause
  \item Differentiate with respect to $w$:
  \begin{equation}
    \nabla_w \mathcal{L}(w,\lambda)=2\Sigma w-2\lambda w = 0.
    \label{eq:lagrange-grad}
  \end{equation}
  \pause
  \item Rearranging gives the eigenvalue equation:
  \begin{equation}
    \Sigma w = \lambda w.
    \label{eq:eigenproblem-pca}
  \end{equation}
  \item Therefore every stationary unit direction of the PCA objective is an eigenvector of the covariance matrix.
  \pause
  \item To identify the maximizing direction, we now compare the objective values of all unit directions.
\end{itemize}
\end{frame}

\begin{frame}{Why the Largest Eigenvalue Matters}
\begin{itemize}
  \item Let $w_j$ be a unit eigenvector of $\Sigma$, so
  \begin{equation}
    \Sigma w_j = \lambda_j w_j.
    \label{eq:eigenpair-pca}
  \end{equation}
  \pause
  \item Evaluate the PCA objective on that eigenvector:
  \begin{equation}
    w_j^\top \Sigma w_j
    = w_j^\top (\lambda_j w_j)
    = \lambda_j \|w_j\|_2^2
    = \lambda_j.
    \label{eq:eigenvalue-variance}
  \end{equation}
  \pause
  \item Therefore, along an eigen-direction, the objective value is exactly its eigenvalue.
  \pause
  \item A larger eigenvalue means a larger projected variance along that direction.
  \pause
  \item If $\lambda_1 \ge \lambda_2 \ge \cdots \ge \lambda_d$, the first principal direction is the eigenvector $w_1$ associated with $\lambda_1$.
  \pause
  \item \footnotesize Example: if $\lambda_1=5$ and $\lambda_2=2$, then the variance along $w_1$ is $5$, while along $w_2$ it is $2$.
\end{itemize}
\end{frame}

\begin{frame}{Why There Are Several Principal Components}
\begin{columns}[T]
\column{0.56\textwidth}
\begin{itemize}
  \item The first component maximizes variance with no other constraint.
  \pause
  \item The second component maximizes variance subject to orthogonality to the first.
  \pause
  \item In general,
  \begin{equation}
    w_k = \arg\max_{\|w\|_2=1,\; w^\top w_j=0\ \forall j<k} w^\top \Sigma w.
    \label{eq:pk}
  \end{equation}
  \pause
  \item This gives orthogonal directions of decreasing variance.
  \pause
  \item \footnotesize Example: $(1,0)^\top(0,1)=0$, so these two directions are orthogonal.
\end{itemize}

\end{frame}

\begin{frame}{Explained Variance and Component Selection}
\begin{itemize}
  \item The explained variance ratio of direction $j$ is
  \begin{equation}
    \mathrm{EVR}_j = \frac{\lambda_j}{\sum_{\ell=1}^d \lambda_\ell}.
    \label{eq:explained-variance-ratio}
  \end{equation}
  \pause
  \item The cumulative explained variance of the first $k$ directions is
  \begin{equation}
    \mathrm{CEV}(k) = \frac{\sum_{j=1}^k \lambda_j}{\sum_{\ell=1}^d \lambda_\ell}.
    \label{eq:cumulative-explained-variance}
  \end{equation}
  \pause
  \item In practice, choose the smallest $k$ that captures a desired fraction such as $90\%$ or $95\%$.
  \pause
  \item \footnotesize Example: if $(\lambda_1,\lambda_2,\lambda_3)=(5,3,2)$, then $\mathrm{EVR}_1=0.5$ and $\mathrm{CEV}(2)=0.8$.
\end{itemize}
\end{frame}

\begin{frame}{Projection, Reconstruction, and Error}
\begin{itemize}
  \item After choosing the first $k$ principal directions, collect them in
  \[
    W_k = [w_1,\dots,w_k] \in \mathbb{R}^{d\times k}.
  \]
  \pause
  \item The low-dimensional representation is
  \begin{equation}
    z = W_k^\top x.
    \label{eq:pca-project-k}
  \end{equation}
  \pause
  \item The reconstruction from that representation is
  \begin{equation}
    \hat{x} = W_k z = W_k W_k^\top x.
    \label{eq:pca-reconstruct}
  \end{equation}
  \pause
  \item If $k$ is small, reconstruction keeps the main structure but discards weaker directions.
  \pause
  \item The reconstruction-error score
  \begin{equation}
    \|x-\hat{x}\|_2^2,
    \label{eq:reconstruction-error}
  \end{equation}
  is useful for denoising and anomaly detection.
  \pause
  \item \footnotesize Example: if $x=(3,1)$ and $w_1=(1,0)$, then $z=w_1^\top x=3$, $\hat{x}=(3,0)$, and $\|x-\hat{x}\|_2^2=1$.
\end{itemize}
\end{frame}

\begin{frame}{When PCA Is Effective}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{PCA often helps when}
\begin{itemize}
  \item features are strongly correlated,
  \pause
  \item many dimensions mostly contain noise,
  \pause
  \item faster distance computation is needed for methods such as k-NN,
  \pause
  \item 2D or 3D visualization is desired.
\end{itemize}

\pause
\column{0.5\textwidth}
\textbf{PCA can hurt when}
\begin{itemize}
  \item low-variance directions still carry target information,
  \pause
  \item interpretability of original features matters more than compression,
  \pause
  \item the data geometry is strongly nonlinear.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Iterative Computation of Principal Components}
\begin{itemize}
  \item For intuition, the leading principal direction can be found by power iteration:
  \begin{equation}
    v^{(t+1)} = \frac{\Sigma v^{(t)}}{\|\Sigma v^{(t)}\|_2}.
    \label{eq:power-iteration}
  \end{equation}
  \pause
  \item Repeated multiplication amplifies the dominant eigen-direction.
  \pause
  \item Later directions can be obtained by removing the first one:
  \begin{equation}
    \Sigma_2 = \Sigma - \lambda_1 w_1 w_1^\top.
    \label{eq:deflation}
  \end{equation}
  \pause
  \item For full PCA in practice, libraries usually prefer SVD instead of repeating this procedure many times.
\end{itemize}
\end{frame}

\begin{frame}{Singular Value Decomposition}
\begin{itemize}
  \item For a centered data matrix $X_c \in \mathbb{R}^{n\times d}$, the singular value decomposition is
  \begin{equation}
    X_c = U S V^\top.
    \label{eq:svd-pca}
  \end{equation}
  \pause
  \item Here $U \in \mathbb{R}^{n\times n}$ and $V \in \mathbb{R}^{d\times d}$ are orthonormal matrices.
  \pause
  \item The diagonal matrix $S$ contains the singular values
  \[
    s_1 \ge s_2 \ge \cdots \ge 0.
  \]
  \pause
  \item In PCA, the columns of $V$ are the directions in feature space, and the singular values measure how strongly the data spread along those directions.
  \pause
  \item \footnotesize Example: if $S=\mathrm{diag}(6,3)$, then the first direction carries twice the singular-value scale of the second.
\end{itemize}
\end{frame}

\begin{frame}{SVD as a Computational Route to PCA}
\begin{itemize}
  \item Instead of first forming
  \[
    \Sigma = \frac{1}{n-1}X_c^\top X_c,
  \]
  libraries often work directly with $X_c$ through \eqref{eq:svd-pca}.
  \pause
  \item Substituting the SVD into $X_c^\top X_c$ gives
  \begin{equation}
    X_c^\top X_c = V S^2 V^\top,
    \label{eq:xtx}
  \end{equation}
  \pause
  \item Therefore
  \[
    \Sigma = \frac{1}{n-1}X_c^\top X_c
           = V \frac{S^2}{n-1} V^\top.
  \]
  \pause
  \item This has the same form as the eigen-decomposition of the covariance matrix.
  \pause
  \item Therefore the columns of $V$ are the principal directions, and
  \[
    \lambda_j = \frac{s_j^2}{n-1}.
  \]
  \pause
  \item Numerical libraries usually prefer this route because it works directly with $X_c$ and avoids an extra covariance-matrix construction.
\end{itemize}
\end{frame}

\begin{frame}{Practical PCA Workflow}
\begin{itemize}
  \item Standardize features first if their scales are very different.
  \pause
  \item Fit PCA on the training split only, after centering the training examples.
  \pause
  \item Apply the same centering vector and PCA directions to the validation and test splits.
  \pause
  \item If reconstruction is needed in the original space, add the training mean back after projection and reconstruction.
  \pause
  \item Choose the number of components by validation or by explained-variance threshold.
  \pause
  \item Then evaluate the reduced representation on the downstream task with the same train/validation/test protocol.
\end{itemize}
\end{frame}

\begin{frame}{Summary}
\begin{itemize}
  \item PCA starts with projection and asks for the direction with maximum projected variance.
  \item The optimization becomes an eigenvalue problem through a constrained quadratic form.
  \item The leading eigenvector gives the first principal direction; the remaining directions follow by orthogonality.
  \item Eigenvalues quantify explained variance and help determine how many components to keep.
  \item PCA also provides reconstruction, which supports compression, denoising, and anomaly scoring.
  \item In practice, PCA can be computed either from the covariance matrix or from SVD.
\end{itemize}
\end{frame}

\begin{frame}{Applications of PCA}
\begin{itemize}
  \item \textbf{Visualization:} reduce high-dimensional embeddings or features to 2D or 3D for inspection.
  \pause
  \item \textbf{Preprocessing for k-NN and other models:} remove redundancy and reduce distance-computation cost.
  \pause
  \item \textbf{Compression and denoising:} keep the main structure with fewer components and discard weak noisy directions.
  \pause
  \item \textbf{Anomaly detection:} use large reconstruction error to flag unusual samples.
  \pause
  \item \textbf{Exploration:} inspect correlated features, dominant variation patterns, and low-dimensional structure before modeling.
\end{itemize}
\end{frame}
