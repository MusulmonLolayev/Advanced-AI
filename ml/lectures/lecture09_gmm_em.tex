\section{Gaussian Mixture Models \& EM}

\begin{frame}{From Hard to Soft Clustering}
\begin{itemize}
  \item $k$-Means assigns every point to exactly one cluster -- a hard, all-or-nothing decision.
  \pause
  \item Points near a cluster boundary are forced to pick a side, even when the evidence is nearly 50/50.
  \pause
  \item A Gaussian Mixture Model (GMM) replaces that hard vote with a \emph{probability} of belonging to each cluster.
  \pause
  \item Clustering becomes a probabilistic model-fitting problem, not just a distance-minimization problem.
\end{itemize}

\pause
\begin{rlintuition}
$k$-Means asks ``which cluster is this point in?'' A GMM asks ``how likely is this point under each cluster's distribution?'' -- and lets the answer be a mixture of both.
\end{rlintuition}
\end{frame}

\begin{frame}{The Mixture Model}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{itemize}
  \item Assume the data was generated by $K$ Gaussian components, each with its own mean and covariance.
  \pause
  \item Every point's density is a weighted sum over all $K$ components:
  \begin{equation}
    p(x) = \sum_{k=1}^{K} \pi_k\, \mathcal{N}(x \mid \mu_k, \Sigma_k).
    \label{eq:gmm-density}
  \end{equation}
  \pause
  \item The mixing weights satisfy $\pi_k \geq 0$ and $\sum_k \pi_k = 1$.
\end{itemize}

\pause
\column{0.5\textwidth}
\centering
\includegraphics[width=\linewidth]{figures/gmm_em/mixture_density_1d.pdf}

{\footnotesize Eq.~\eqref{eq:gmm-density} in 1D: two equal-weight components at $\mu_1=0,\mu_2=10$; the mixture (green) is their weighted sum.}
\end{columns}
\end{frame}

\begin{frame}{Latent Variables: Which Component Generated This Point?}
\begin{itemize}
  \item Introduce an unobserved (latent) indicator $z_n \in \{1,\dots,K\}$ for each point: the component that generated $x_n$.
  \pause
  \item If we knew every $z_n$, fitting the model would be trivial: split the data by label and fit each Gaussian separately.
  \pause
  \item If we knew $\{\pi_k,\mu_k,\Sigma_k\}$, inferring $z_n$ would also be easy: compare densities under each component.
  \pause
  \item The catch: we know neither. Both must be estimated together.
\end{itemize}

\pause
\begin{rlpitfall}
This is a chicken-and-egg problem -- it cannot be solved in one shot by a closed-form formula. EM solves it by alternating between the two easy sub-problems.
\end{rlpitfall}
\end{frame}

\begin{frame}{The EM Idea}
\begin{itemize}
  \item \textbf{E-step (Expectation):} fix the current parameters, compute the probability that each point belongs to each component.
  \pause
  \item \textbf{M-step (Maximization):} fix those probabilities, re-estimate the parameters that best explain the (softly labeled) data.
  \pause
  \item Repeat E-step and M-step until the parameters stop changing.
  \pause
  \item Each round only ever solves one of the two easy sub-problems from the previous slide -- never both at once.
\end{itemize}
\end{frame}

\begin{frame}{E-step: Responsibilities}
\begin{itemize}
  \item Define the \emph{responsibility} $\gamma_{nk}$: the posterior probability that component $k$ generated point $x_n$.
  \pause
  \item By Bayes' rule, using the current parameter estimates:
  \begin{equation}
    \gamma_{nk} = \frac{\pi_k\, \mathcal{N}(x_n \mid \mu_k, \Sigma_k)}{\sum_{j=1}^{K} \pi_j\, \mathcal{N}(x_n \mid \mu_j, \Sigma_j)}.
    \label{eq:responsibility}
  \end{equation}
  \pause
  \item For every $n$: $\sum_{k=1}^{K} \gamma_{nk} = 1$ -- a soft assignment that sums to one, not a hard label.
\end{itemize}

\pause
\begin{rldefinition}
$\gamma_{nk}$ is a \emph{soft assignment}: how much of point $n$'s ``mass'' is explained by component $k$, given the current model.
\end{rldefinition}
\end{frame}

\begin{frame}{M-step: Re-estimating the Parameters}
\begin{itemize}
  \item Treat $\gamma_{nk}$ as a soft count and re-fit each component's parameters:
  \begin{equation}
    N_k = \sum_{n=1}^{N} \gamma_{nk}, \qquad
    \mu_k = \frac{1}{N_k}\sum_{n=1}^{N} \gamma_{nk}\, x_n,
    \label{eq:m-step-mean}
  \end{equation}
  \pause
  \begin{equation}
    \Sigma_k = \frac{1}{N_k}\sum_{n=1}^{N} \gamma_{nk}\,(x_n-\mu_k)(x_n-\mu_k)^{\!\top},
    \qquad
    \pi_k = \frac{N_k}{N}.
    \label{eq:m-step-cov}
  \end{equation}
  \pause
  \item $N_k$ is the \emph{effective} number of points assigned to component $k$ -- it need not be an integer.
\end{itemize}
\end{frame}

\begin{frame}{The EM Algorithm, End to End}
\begin{itemize}
  \item Initialize $\{\pi_k,\mu_k,\Sigma_k\}$ (e.g.\ from a $k$-means run, or randomly).
  \pause
  \item Repeat until convergence:
  \begin{itemize}
    \item \textbf{E-step:} compute $\gamma_{nk}$ for all $n,k$ using Eq.~\eqref{eq:responsibility}.
    \item \textbf{M-step:} update $\{\pi_k,\mu_k,\Sigma_k\}$ using Eqs.~\eqref{eq:m-step-mean}--\eqref{eq:m-step-cov}.
  \end{itemize}
  \pause
  \item Convergence check: the log-likelihood stops increasing (or increases by less than a small tolerance).
\end{itemize}
\end{frame}

\begin{frame}{Worked Numerical Example: Setup}
\small
\begin{itemize}
  \item Four 1D points: $x \in \{1, 2, 8, 9\}$ -- two natural clusters, near $1.5$ and near $8.5$.
  \item Two components, equal weight $\pi_1=\pi_2=0.5$, shared known $\sigma=3$.
  \item Deliberately bad initialization: $\mu_1^{(0)}=0$, $\mu_2^{(0)}=10$.
  \pause
  \item Because $\sigma$ is shared, Eq.~\eqref{eq:responsibility} simplifies to a logistic form:
  \[
    \gamma_{n1} = \frac{1}{1+\exp\!\left(-\dfrac{(x_n-\mu_2)^2-(x_n-\mu_1)^2}{2\sigma^2}\right)}.
  \]
\end{itemize}
\end{frame}

\begin{frame}{Worked Numerical Example: M-step}
\small
\begin{itemize}
  \item Effective counts: $N_1 = 0.9884+0.9656+0.0344+0.0116 = 2.000$, and by symmetry $N_2 = 2.000$.
  \pause
  \item Updated mean of component 1, by Eq.~\eqref{eq:m-step-mean}:
  \[
    \mu_1^{(1)} = \frac{0.9884(1)+0.9656(2)+0.0344(8)+0.0116(9)}{2.000} \approx 1.650.
  \]
  \pause
  \item Updated mean of component 2:
  \[
    \mu_2^{(1)} = \frac{0.0116(1)+0.0344(2)+0.9656(8)+0.9884(9)}{2.000} \approx 8.350.
  \]
  \pause
  \item One iteration moved $\mu_1: 0\to 1.65$ and $\mu_2: 10\to 8.35$ -- both much closer to the true cluster centers near $1.5$ and $8.5$.
\end{itemize}
\end{frame}

\begin{frame}{Why EM Is Guaranteed to Improve}
\begin{itemize}
  \item Each E-step computes responsibilities that exactly match the current parameters.
  \pause
  \item Each M-step then chooses the parameters that maximize the (responsibility-weighted) likelihood.
  \pause
  \item Both steps can only increase, never decrease, the data log-likelihood.
  \pause
  \item Result: log-likelihood rises monotonically and the algorithm converges to a (local) maximum.
\end{itemize}
\end{frame}

\begin{frame}{Common Pitfalls}
\begin{rlpitfall}
\textbf{Singular covariances:} if a component's variance collapses onto a single point, its density (and the log-likelihood) blows up to infinity. Guard with a minimum-covariance floor (\texttt{reg\_covar}).
\end{rlpitfall}
\pause
\begin{itemize}
  \item \textbf{Initialization sensitivity:} EM only finds a \emph{local} optimum -- different starting points can converge to different solutions. Run several initializations (often seeded from $k$-means) and keep the best.
  \pause
  \item \textbf{Label switching:} components have no inherent order; component 1 in one run may match component 2 in another.
  \pause
  \item \textbf{Choosing $K$:} log-likelihood alone always improves with more components -- use a held-out criterion (BIC/AIC) or domain knowledge instead.
\end{itemize}
\end{frame}

\begin{frame}{Think About This}
\begin{rlquestion}
If you shrink a shared, isotropic GMM's $\sigma$ toward $0$, the responsibilities in Eq.~\eqref{eq:responsibility} sharpen toward $0$ or $1$. What clustering algorithm do you recover in that limit, and what does the model lose by making that assumption?
\end{rlquestion}
\pause
\begin{itemize}
  \item Answer: $k$-Means -- hard assignment, equal spherical clusters, no notion of cluster shape or overlap.
  \pause
  \item The GMM's extra flexibility (covariance shape, soft assignment) is exactly what is thrown away in that limit.
\end{itemize}
\end{frame}

\begin{frame}{Key Hyperparameters}
\begin{itemize}
  \item \textbf{\texttt{n\_components}} $K$: number of mixture components. Too few underfits the clusters; too many overfits and risks singular components.
  \pause
  \item \textbf{\texttt{covariance\_type}}: spherical, diagonal, tied, or full -- controls how much shape each component is allowed.
  \pause
  \item \textbf{\texttt{n\_init}}: number of random restarts; mitigates sensitivity to local optima.
  \pause
  \item \textbf{\texttt{reg\_covar}}: small constant added to the covariance diagonal to avoid singularities.
  \pause
  \item \textbf{\texttt{tol}} / \textbf{\texttt{max\_iter}}: convergence tolerance on log-likelihood and the iteration budget.
\end{itemize}
\end{frame}

\begin{frame}{Summary}
\begin{itemize}
  \item A GMM models data as a weighted sum of Gaussian components and replaces hard cluster labels with soft responsibilities.
  \item EM alternates between two easy sub-problems: E-step computes responsibilities $\gamma_{nk}$ from the current parameters (Eq.~\eqref{eq:responsibility}); M-step refits $\{\pi_k,\mu_k,\Sigma_k\}$ from those responsibilities (Eqs.~\eqref{eq:m-step-mean}--\eqref{eq:m-step-cov}).
  \item Each EM round provably never decreases the data log-likelihood, but only reaches a local optimum -- initialization matters.
  \item $k$-Means is the hard-assignment, equal-spherical-cluster limit of a GMM.
  \item Singular covariances, label switching, and choosing $K$ are the recurring practical pitfalls.
\end{itemize}
\end{frame}
